  const qs = sampleId !== undefined && sampleId !== null && !Number.isNaN(sampleId)
    ? `?sampleId=${encodeURIComponent(String(sampleId))}`
    : "";

}
AFTER:
export async function getReviewQueuePage(
  sampleId?: number,
  view?: string): Promise&lt;ReviewQueuePageResponse&gt; {
  const params = new URLSearchParams()
  if (sampleId !== undefined && !Number.isNaN(sampleId))
    params.set('sampleId', String(sampleId))
  if (view &amp;&amp; view !== 'dept')
    params.set('view', view)
  const qs = params.toString() ? `?${params}` : ""
  return await get&lt;ReviewQueuePageResponse&gt;(
    `/api/v1/reviews/queue${qs}`)
}
