    r.[Sample_id],                 -- 24
    r.[Review_type],               -- 25
    r.[CRO_email],                 -- 26

    r.[Appeal_description],        -- 36

    (SELECT TOP 1 r2.[Review_id]
     FROM dbo.[02_CORE_02_Reviews] r2 WITH (NOLOCK)
     WHERE r2.[Customer_number] = r.[Customer_number]
       AND r2.[Review_id] != r.[Review_id]
       AND (r2.[Cancelled] IS NULL OR r2.[Cancelled] = 0)
       AND r2.[Review_finalized_date] = (
         SELECT MAX(r3.[Review_finalized_date])
         FROM dbo.[02_CORE_02_Reviews] r3 WITH (NOLOCK)
         WHERE r3.[Customer_number] = r.[Customer_number]
           AND r3.[Review_id] != r.[Review_id]
           AND (r3.[Cancelled] IS NULL OR r3.[Cancelled] = 0)
       )
    ) AS [Prior_CAS_review_id],
    (SELECT MAX(r2.[Review_finalized_date])
     FROM dbo.[02_CORE_02_Reviews] r2 WITH (NOLOCK)
     WHERE r2.[Customer_number] = r.[Customer_number]
       AND r2.[Review_id] != r.[Review_id]
       AND (r2.[Cancelled] IS NULL OR r2.[Cancelled] = 0)
       AND r2.[Review_finalized_date] IS NOT NULL
    ) AS [Prior_CAS_review_date],
    r.[Review_initial_approval_date],
    r.[Mentor_name],
    r.[Mentor_review_date]
FROM dbo.[02_CORE_02_Reviews] AS r WITH (NOLOCK)
WHERE (r.[Cancelled] IS NULL OR r.[Cancelled] = 0)
  AND r.[Review_id] = @id
ORDER BY
    ISNULL(r.[Review_approval_date], ISNULL(r.[Review_finalized_date], ISNULL(r.[Completed_date], ISNULL(r.[Review_distributed_date], ISNULL(r.[Start_date], '1900-01-01'))))) DESC,
    r.[Review_id] DESC;
