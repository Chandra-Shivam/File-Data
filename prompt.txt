WITH CurrentUser AS (
    SELECT [CAS_user_name], [CAS_user_id]
    FROM dbo.[03_LIBRARY_05_CAS_Users] WITH (NOLOCK)
    WHERE [CAS_user_id] = @empNumber
      AND ([CAS_user_active] = 1 
           OR [CAS_user_active] IS NULL)

TeamMembers AS (
    SELECT u.[CAS_user_name]
    FROM dbo.[03_LIBRARY_05_CAS_Users] u WITH (NOLOCK)
    INNER JOIN CurrentUser cu 
        ON u.[CAS_user_manager] = cu.[CAS_user_name]
    WHERE [CAS_user_active] = 1 
       OR [CAS_user_active] IS NULL
)
SELECT TOP (250)
    r.[Review_id],
    r.[Sample_id],
    r.[eCIF_number],
    r.[Customer_name],
    r.[CRO_name],
    r.[Relationship_mgr_name],
    r.[Portfolio_mgr_name],
    r.[TTBA_exposure],
    r.[Bank_PD],
    r.[CAS_PD],
    r.[Completed_date],
    r.[Review_finalized_date],
    r.[Review_approval_date],
    r.[Review_distributed_date],
    r.[Start_date],
    agg.[CommittedExposure] AS [AccountsCommittedExposure]
FROM dbo.[02_CORE_02_Reviews] AS r WITH (NOLOCK)
LEFT JOIN (
    SELECT a.[Review_id], 
      SUM(COALESCE(a.[Commitment], 0)) AS [CommittedExposure]
    FROM dbo.[02_CORE_04_Accounts] AS a WITH (NOLOCK)
    GROUP BY a.[Review_id]
) AS agg ON agg.[Review_id] = r.[Review_id]
WHERE (r.[Cancelled] IS NULL OR r.[Cancelled] = 0)
  AND (@sampleId IS NULL OR r.[Sample_id] = @sampleId)
  AND (
    @view = 'dept'
    OR (
      @view = 'my' AND (
        EXISTS (SELECT 1 FROM CurrentUser cu 
          WHERE cu.[CAS_user_name] = r.[CRO_name])
        OR EXISTS (SELECT 1 FROM CurrentUser cu 
          WHERE cu.[CAS_user_name] = 
            r.[Relationship_mgr_name])
        OR EXISTS (SELECT 1 FROM CurrentUser cu 
          WHERE cu.[CAS_user_name] = 
            r.[Portfolio_mgr_name])
      )
    )
    OR (
      @view = 'team' AND (
        EXISTS (SELECT 1 FROM TeamMembers tm 
          WHERE tm.[CAS_user_name] = r.[CRO_name])
        OR EXISTS (SELECT 1 FROM TeamMembers tm 
          WHERE tm.[CAS_user_name] = 
            r.[Relationship_mgr_name])
        OR EXISTS (SELECT 1 FROM TeamMembers tm 
          WHERE tm.[CAS_user_name] = 
            r.[Portfolio_mgr_name])
      )
    )
  )
ORDER BY
    ISNULL(r.[Review_approval_date], 
      ISNULL(r.[Review_finalized_date], 
        ISNULL(r.[Completed_date], 
          ISNULL(r.[Review_distributed_date], 
            ISNULL(r.[Start_date], '1900-01-01'))))) DESC,
    r.[Review_id] DESC;

export async function getReviewQueuePage(

export async function getReviewQueuePage(
  sampleId?: number, 
  view?: string): Promise<ReviewQueuePageResponse> {
  const params = new URLSearchParams()
  if (sampleId !== undefined && !Number.isNaN(sampleId))
    params.set('sampleId', String(sampleId))
  if (view) params.set('view', view)
  const qs = params.toString() ? `?${params.toString()}` : ""
  return await get<ReviewQueuePageResponse>(
    `/api/v1/reviews/queue${qs}`)
}
